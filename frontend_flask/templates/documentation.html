Getting Started

Welcome to AetherMind documentation. Learn how to integrate continuous learning AGI into your application.

What is AetherMind?

AetherMind is a continuous learning AGI that adapts to your domain through experience, not prompt engineering. Built with axiom-based learning, infinite episodic memory, and self-modification capabilities.

Quick Start

Get started in 3 steps:

1. Get Your API Key

Sign up and generate your API key from the dashboard.

2. Install SDK

pip install aethermind-sdk
# or
npm install @aethermind/sdk

3. Make Your First Request

from aethermind import Client

client = Client(api_key="am_live_YOUR_KEY")
response = client.chat("Hello AetherMind!")
print(response)

Authentication

AetherMind supports three authentication methods to fit any use case:

API Keys

Long-lived credentials for server-to-server authentication

Recommended

JWT Tokens

Short-lived, stateless authentication for web/mobile apps

Best for Web

Sessions

Cookie-based authentication for traditional web applications

Legacy

API Key Authentication

Generate API Key

Create a new API key from your dashboard or via API:

from orchestrator.auth_manager import AuthManager, UserRole

auth = AuthManager()
api_key = auth.generate_api_key(
    user_id="your_github_username",
    role=UserRole.PRO,
    metadata={"email": "you@example.com"}
)

print(f"Your API key: {api_key}")
# Output: am_live_XXXXXXXXXXXXX

Use API Key

Include your API key in the Authorization header:

curl -X POST https://api.aethermind.ai/v1/chat \
  -H "Authorization: ApiKey am_live_YOUR_KEY" \
  -H "Content-Type: application/json" \
  -d '{"message": "Hello AetherMind"}'

Security Warning

Never expose API keys in client-side code (frontend JavaScript, mobile apps). Always call AetherMind from your backend.

JWT Token Authentication

JWT tokens are ideal for web and mobile applications. They expire after a set duration (default 1 hour).

Generate JWT Token

jwt_token = auth.generate_jwt_token(
    user_id="user123",
    role=UserRole.PRO,
    expires_in=3600  # 1 hour
)

# Return to client
return {"token": jwt_token}

Use JWT Token

curl -X POST https://api.aethermind.ai/v1/chat \
  -H "Authorization: Bearer eyJhbGciOiJIUzI1NiIsInR5cCI6IkpXVCJ9..." \
  -H "Content-Type: application/json" \
  -d '{"message": "Hello"}'

Platform Integrations

Python

import requests

class AetherMindClient:
    def __init__(self, api_key):
        self.api_key = api_key
        self.base_url = "https://api.aethermind.ai/v1"
    
    def chat(self, message):
        response = requests.post(
            f"{self.base_url}/chat",
            headers={"Authorization": f"ApiKey {self.api_key}"},
            json={"message": message}
        )
        return response.json()

# Usage
client = AetherMindClient("am_live_YOUR_KEY")
result = client.chat("Explain quantum computing")
print(result)

Next.js (TypeScript)

// lib/aethermind.ts
export class AetherMindClient {
  private apiKey: string;
  private baseUrl = 'https://api.aethermind.ai/v1';

  constructor(apiKey: string) {
    this.apiKey = apiKey;
  }

  async chat(message: string): Promise<any> {
    const response = await fetch(`${this.baseUrl}/chat`, {
      method: 'POST',
      headers: {
        'Authorization': `ApiKey ${this.apiKey}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({ message })
    });
    return response.json();
  }
}

// app/api/chat/route.ts
import { AetherMindClient } from '@/lib/aethermind';

export async function POST(request: Request) {
  const client = new AetherMindClient(process.env.AETHERMIND_API_KEY!);
  const { message } = await request.json();
  const result = await client.chat(message);
  return Response.json(result);
}

React

// src/components/Chat.jsx
import { useState } from 'react';

export default function Chat() {
  const [message, setMessage] = useState('');
  const [response, setResponse] = useState('');
  
  const sendMessage = async () => {
    // Call your backend API (never expose API key in frontend)
    const res = await fetch('/api/chat', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ message })
    });
    
    const data = await res.json();
    setResponse(data.response);
  };
  
  return (
    <div>
      <input value={message} onChange={(e) => setMessage(e.target.value)} />
      <button onClick={sendMessage}>Send</button>
      <div>{response}</div>
    </div>
  );
}

Node.js / Express

const express = require('express');
const axios = require('axios');

const app = express();
app.use(express.json());

app.post('/api/chat', async (req, res) => {
  try {
    const response = await axios.post(
      'https://api.aethermind.ai/v1/chat',
      { message: req.body.message },
      {
        headers: {
          'Authorization': `ApiKey ${process.env.AETHERMIND_API_KEY}`
        }
      }
    );
    res.json(response.data);
  } catch (error) {
    res.status(500).json({ error: error.message });
  }
});

app.listen(3000);

Flutter

// lib/services/aethermind.dart
import 'dart:convert';
import 'package:http/http.dart' as http;

class AetherMindClient {
  final String apiKey;
  
  AetherMindClient(this.apiKey);
  
  Future<Map> chat(String message) async {
    final response = await http.post(
      Uri.parse('https://api.aethermind.ai/v1/chat'),
      headers: {
        'Authorization': 'ApiKey $apiKey',
        'Content-Type': 'application/json'
      },
      body: jsonEncode({'message': message})
    );
    return jsonDecode(response.body);
  }
}

Security Best Practices

DON'T

  • ❌ Hardcode API keys in client-side code
  • ❌ Commit API keys to version control
  • ❌ Share API keys publicly
  • ❌ Use production keys in development

DO

  • ✅ Store keys in environment variables
  • ✅ Use different keys for dev/staging/prod
  • ✅ Call AetherMind from your backend
  • ✅ Rotate keys regularly
  • ✅ Use JWT tokens for client apps
